actor CheckLiveRound:ActionItem{
	states{
	pickup:
		TNT1 A 0{
			if(countinv("FragOut")>0){  
				A_GiveInventory("HDGrenadeToss",1);
				A_TakeInventory("FragOut");
				A_TakeInventory("SpoonOff");
			}
			if(countinv("BFGCycled")>0){  
				A_DropInventory("BFG9k");
			}
			if(countinv("ZM66Heat")>26){
				A_DropInventory("ZM66Rifle");
				A_DropInventory("ZM66GrenadeRifle");
				A_DropInventory("ZM66SemiRifle");
			}
		}
		stop
	}
}

actor NulledWeapon:InventoryFlag{}
actor NullWeapon:Weapon{
	+weapon.wimpy_weapon
	+weapon.cheatnotweapon
	+nointeraction
	+weapon.noalert
	+inventory.untossable
	height 0
	radius 0
	states{
	spawn:
		TNT1 A 0
		stop
	select:
		TNT1 A 0{
			A_TakeInventory("IsWeaponReady");
			A_TakeInventory("IsWeaponShort");
			A_TakeInventory("IsWeaponLong");
			A_TakeInventory("IsWeaponBFG");
			A_TakeInventory("PowerFrightener",1);
			A_GiveInventory("NulledWeapon",1);
			A_SetCrosshair(21);
		}
		TNT1 A 0 A_Raise
		wait
	deselect:
		TNT1 A 0 A_SetCrosshair(21)
		TNT1 A 0 A_Lower
		wait
	ready:
		TNT1 A 1{
			A_SetCrosshair(21);
			A_WeaponReady(WRF_NOFIRE);
			A_TakeInventory("WeaponBusy");
		}loop
	fire:
		TNT1 A 1
		goto ready
	}
}
