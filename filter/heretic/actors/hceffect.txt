ACTOR SDSmokeSmall:HDSmokeSmall replaces HDSmokeSmall{
	states{
	spawn:
		FX04 E 2 nodelay A_SetScale((random(0,1)*2-1)*scalex,scaley)
		FX04 E 0 A_SetTranslucent(0.7,0)
		FX04 E 0 A_ChangeVelocity(0,0,2)
		FX04 EEFFGGG 5{
			A_FadeOut(0.1);
			A_ScaleVelocity(0.7);
			if(scalex<2){A_SetScale(scalex+0.2);}
		}stop
	}
}
