//to check for D1 replacements if D2 resources not found
//most of these are already built into the main code and not here

actor Shardangel:HDActor replaces ArchAngel{
	+ismonster +friendly states{
	spawn:
		TNT1 A 0 nodelay{
			if(!CheckLump("VILER0")){return state("spawn1");}
			A_SpawnItemEx("ArchAngel",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION|SXF_TRANSFERPOINTERS);
			return state("null");
		}stop
	spawn1:
		TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0{
			A_SpawnItemEx("BFGVileShard",random(-3,3),random(-3,3),random(1,6),random(1,3)*10,0,random(1,12),random(0,360),SXF_SETMASTER);
		}
		TNT1 A 4
		TNT1 A 0 A_AlertMonsters
		stop
	}
}
